คู่มือการตั้งค่ามอด Baby Player (Config)

เมื่อผู้เล่นโตเร็วเกินไป หรือเราต้องการให้พวกเขาเป็นเด็กตลอดไป คุณสามารถปรับแต่งค่าเหล่านี้ได้ในไฟล์ Config ครับ

💡 เทคนิคด่วน (Quick Tips):

babyAge = -4000 // ถ้าต้องการให้โตไวขึ้น ให้ปรับค่าให้น้อยลง
babyAge = -2147483647 // ถ้าต้องการให้เป็นเด็กตลอดไป (ไม่โตเลย)

ไฟล์ตั้งค่าฉบับเต็ม (.toml)

[babyplayers_common]
  # อายุเริ่มต้นของตัวละครเด็ก (ตั้งเป็น 0 เพื่อปิดระบบตัวเล็ก)
  babyAge = -2147483647

  # อนุญาตให้เด็กเติบโตเป็นผู้ใหญ่ได้ (true = เปิด / false = ปิด)
  doAging = true

  # กำหนดให้ผู้เล่นเกิดใหม่ในร่างเด็กเสมอ
  respawnAsBaby = true

  # กำหนดให้ร่างเด็กสามารถขุดบล็อกได้ 
  # (บล็อกที่แตกทันที เช่น หญ้า จะไม่มีผลจากค่านี้)
  babiesBreakBlocks = true

  # กำหนดให้ร่างเด็กสร้างความเสียหาย (โจมตี) เบากว่าผู้ใหญ่ 4 เท่า
  babiesDealLessDamage = true

  # ถ้าตั้งเป็น true เด็กจะไม่สามารถใช้อาวุธระยะไกลได้ (ธนู, หน้าไม้)
  babiesCantUseBows = true
